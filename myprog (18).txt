      LDR R0,. InputNum
      CMP R0, #1
      BEQ Addition
      BNE Subtraction
Addition:
      LDR R2,. InputNum
      LDR R3,. InputNum
      ADD R4, R2, R3
      B Display
Subtraction:
      LDR R5,. InputNum
      LDR R6,. InputNum
      SUB R7, R5, R6
      B Display
Display:
      MOV R6, R10
